015    package com.cloudera.lib.servlet;
016    
017    import org.slf4j.MDC;
018    
019    import javax.servlet.Filter;
020    import javax.servlet.FilterChain;
021    import javax.servlet.FilterConfig;
022    import javax.servlet.ServletException;
023    import javax.servlet.ServletRequest;
024    import javax.servlet.ServletResponse;
025    import javax.servlet.http.HttpServletRequest;
026    import java.io.IOException;
027    import java.security.Principal;
028    
029    public class MDCFilter implements Filter {
030    
031      @Override
032      public void init(FilterConfig config) throws ServletException {
033      }
034    
035      @Override
036      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
037        throws IOException, ServletException {
038        try {
039          MDC.clear();
040          String hostname = HostnameFilter.get();
041          if (hostname != null) {
042            MDC.put("hostname", HostnameFilter.get());
043          }
044          Principal principal = ((HttpServletRequest) request).getUserPrincipal();
045          String user = (principal != null) ? principal.getName() : null;
046          if (user != null) {
047            MDC.put("user", user);
048          }
049          MDC.put("method", ((HttpServletRequest) request).getMethod());
050          MDC.put("path", ((HttpServletRequest) request).getPathInfo());
051          chain.doFilter(request, response);
052        }
053        finally {
054          MDC.clear();
055        }
056      }
057    
058      @Override
059      public void destroy() {
060      }
061    }
