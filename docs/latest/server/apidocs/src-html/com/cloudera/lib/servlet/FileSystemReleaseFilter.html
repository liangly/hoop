015    package com.cloudera.lib.servlet;
016    
017    import com.cloudera.lib.service.Hadoop;
018    import org.apache.hadoop.fs.FileSystem;
019    
020    import javax.servlet.Filter;
021    import javax.servlet.FilterChain;
022    import javax.servlet.FilterConfig;
023    import javax.servlet.ServletException;
024    import javax.servlet.ServletRequest;
025    import javax.servlet.ServletResponse;
026    import java.io.IOException;
027    
028    public abstract class FileSystemReleaseFilter implements Filter {
029      private static final ThreadLocal<FileSystem> FILE_SYSTEM_TL = new ThreadLocal<FileSystem>();
030    
031      @Override
032      public void init(FilterConfig filterConfig) throws ServletException {
033      }
034    
035      @Override
036      public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain)
037        throws IOException, ServletException {
038        try {
039          filterChain.doFilter(servletRequest, servletResponse);
040        }
041        finally {
042          FileSystem fs = FILE_SYSTEM_TL.get();
043          if (fs != null) {
044            FILE_SYSTEM_TL.remove();
045            getHadoop().releaseFileSystem(fs);
046          }
047        }
048      }
049    
050      @Override
051      public void destroy() {
052      }
053    
054      public static void setFileSystem(FileSystem fs) {
055        FILE_SYSTEM_TL.set(fs);
056      }
057    
058      protected abstract Hadoop getHadoop();
059    
060    }
