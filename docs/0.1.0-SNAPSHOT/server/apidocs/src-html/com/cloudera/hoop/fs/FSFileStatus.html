015    package com.cloudera.hoop.fs;
016    
017    import com.cloudera.hoop.HoopServer;
018    import com.cloudera.lib.service.Hadoop;
019    import org.apache.hadoop.fs.FileStatus;
020    import org.apache.hadoop.fs.FileSystem;
021    import org.apache.hadoop.fs.Path;
022    
023    import java.io.IOException;
024    import java.util.Map;
025    
026    public class FSFileStatus implements Hadoop.FileSystemExecutor<Map> {
027      private Path path;
028    
029      public FSFileStatus(String path) {
030        this.path = new Path(path);
031      }
032    
033      @Override
034      public Map execute(FileSystem fs) throws IOException {
035        FileStatus status = fs.getFileStatus(path);
036        return FSUtils.fileStatusToJSON(status, HoopServer.get().getBaseUrl());
037      }
038    
039    }
